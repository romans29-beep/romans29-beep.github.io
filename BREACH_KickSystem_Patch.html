function normalizeName(name) {
  return (name || "")
    .toUpperCase()
    .normalize("NFKD")
    .replace(/[@4]/g,"A")
    .replace(/[3]/g,"E")
    .replace(/[1!|]/g,"I")
    .replace(/[0]/g,"O")
    .replace(/[5$]/g,"S")
    .replace(/[^A-Z]/g,"")
    .replace(/(.)\1+/g,"$1");
}
